# 切换到项目根目录，确保后续命令都在正确路径下执行
cd D:\graduation-project


# 解析当前虚拟环境里的 Python 可执行文件路径，供后续命令统一复用
$pythonExe = (Resolve-Path .\.venv\Scripts\python.exe).Path


# 解析当前虚拟环境里的 uvicorn 可执行文件路径，供后续后端启动命令复用
$uvicornExe = (Resolve-Path .\.venv\Scripts\uvicorn.exe).Path


# 生成带时间戳的演示数据集名称，避免重复执行脚本时发生同名冲突
$demoDatasetName = "demo_template_{0}" -f (Get-Date -Format "yyyyMMdd_HHmmss")


# 启动 PostgreSQL 数据库容器，供后端读写业务数据
docker compose -f deploy/docker-compose.yml up -d postgres


# 执行 Alembic 数据库迁移，创建并更新当前系统所需的数据表结构
& $pythonExe -m alembic -c backend/alembic.ini upgrade head


# 初始化管理员账号，便于首次登录后台系统
& $pythonExe backend/scripts/init_admin.py


# 导入一份示例交易模板数据（必要列在前、可选列在后），确保总览页、分析中心页和算法雷达页有可展示内容
& $pythonExe backend/scripts/import_data.py --file-path .\web\public\trading_import_template.csv --dataset-name $demoDatasetName


# 以下两个启动命令都是长驻服务，建议分别在独立终端执行


# 在当前终端启动 FastAPI 后端开发服务，并监听本机 8200 端口
& $uvicornExe app.main:app --app-dir backend --host 127.0.0.1 --port 8200 --reload


# 在另一个终端切换到 Web 前端工程目录，准备启动 Vue 开发服务器
cd D:\graduation-project\web


# 在前端终端启动开发服务，默认运行在 4173 端口
npm run dev


# 在算法雷达页中，可切换 `persistent_segment_tree` 与 `t_digest` 查看区间第 K 大结果


# 以管理员身份登录后，可在系统总览页使用用户管理面板编辑普通用户、启用/禁用账号


# 在浏览器中打开系统首页，未登录状态会自动进入登录流程
start http://127.0.0.1:4173/
