# 切换到项目根目录，确保整个初始化流程从统一位置开始
cd D:\graduation-project


# 创建项目专用的 Python 虚拟环境，并指定使用 Python 3.13
uv venv .venv --python 3.13


# 根据 pyproject.toml 安装后端所需的 Python 依赖
uv sync


# 解析当前虚拟环境里的 Python 可执行文件路径，供后续 CMake 和 pybind11 构建使用
$pythonExe = (Resolve-Path .\.venv\Scripts\python.exe).Path


# 获取 pybind11 的 CMake 配置目录，确保 algo-module 在新环境下也能正确配置
$pybind11CMakeDir = & $pythonExe -m pybind11 --cmakedir


# 进入前端目录，准备安装 Node.js 依赖
cd frontend


# 安装前端项目依赖包
npm install


# 返回项目根目录，继续执行后端和数据库相关初始化
cd ..


# 配置 C++ algo-module 工程，生成 pybind11 Python 绑定所需的构建文件
cmake -S algo-module -B algo-module/build -DPython_EXECUTABLE="$pythonExe" -Dpybind11_DIR="$pybind11CMakeDir"


# 编译 C++ 算法模块和 Python 绑定模块，供后端算法接口直接调用
cmake --build algo-module/build


# 复制环境变量模板，生成本地运行所需的 .env 配置文件
Copy-Item .env.template .env -Force


# 复制前端环境变量模板，确保 Vite 代理地址与开发端口可显式配置
Copy-Item .\frontend\.env.template .\frontend\.env -Force


# 启动 PostgreSQL 数据库容器，供系统持久化存储用户和交易数据
docker compose -f deploy/docker-compose.yml up -d postgres


# 执行数据库迁移，应用当前版本的所有表结构和清理逻辑
.\.venv\Scripts\python.exe -m alembic -c backend/alembic.ini upgrade head


# 初始化默认管理员账号，便于系统首次登录和管理
.\.venv\Scripts\python.exe backend/scripts/init_admin.py


# 导入一份示例交易模板数据（必要列在前、可选列在后），用于快速验证上传与分析链路是否正常
.\.venv\Scripts\python.exe backend/scripts/import_data.py --file-path .\frontend\public\trading_import_template.csv --dataset-name demo_template


# 运行 C++ 单测与 Python 绑定 smoke test，验证 algo-module 构建产物可用
ctest --test-dir algo-module/build --output-on-failure


# 运行交易算法接口测试，验证区间最大值、主席树第 K 大以及 t-digest 近似链路
.\.venv\Scripts\python.exe backend/tests/test_algo_trading.py


# 运行 t-digest 误差与 rank error 对拍测试，验证近似方案的边界行为
.\.venv\Scripts\python.exe backend/tests/test_tdigest_range_kth.py


# 运行交易分析测试，验证分析接口与指标链路正常
.\.venv\Scripts\python.exe backend/tests/test_trading_analysis.py


# 运行导入链路与权限测试，验证列头映射、可选 amount 和导入后查询流程正常
.\.venv\Scripts\python.exe backend/tests/test_database_pipeline.py


# 进入前端目录，执行生产构建校验，验证页面改动与类型检查通过
cd D:\graduation-project\frontend


# 运行前端构建，验证上传页和分析页改动可正常编译
npm run build


# 返回项目根目录，继续执行开发服务启动命令
cd ..


# 以下两个启动命令都是长驻服务，建议分别在独立终端执行


# 在当前终端启动 FastAPI 后端开发服务，并监听本机 8200 端口
.\.venv\Scripts\uvicorn.exe app.main:app --app-dir backend --host 127.0.0.1 --port 8200 --reload


# 在另一个终端切换到前端工程目录，准备启动 Vue 开发服务器
cd D:\graduation-project\frontend


# 在前端终端启动开发服务，默认提供 127.0.0.1:4173 的管理系统界面
npm run dev


# 在浏览器中打开登录页，进入系统进行功能验证
start http://127.0.0.1:4173/login


# 安装项目级 benchmark 所需的可选依赖（matplotlib / psutil / locust）
uv sync --extra benchmark


# 运行高频查询效率基准，结果与图表将写入根目录 benchmarks/query_efficiency/
python benchmarks/run_all.py --suite query_efficiency --sample all


# 运行第 K 大三方案对比基准，结果与图表将写入根目录 benchmarks/kth_comparison/
python benchmarks/run_all.py --suite kth_comparison --sample all


# 运行平台稳定性与安全性压测前，请先设置 Locust 所需环境变量
$env:BENCHMARK_BASE_URL = "http://127.0.0.1:8200"
$env:BENCHMARK_USERNAME = "admin"
$env:BENCHMARK_PASSWORD = "admin123456"
$env:BENCHMARK_IMPORT_RUN_ID = "1"
$env:BENCHMARK_SAMPLE = "smoke"


# 运行平台压测，结果与图表将写入根目录 benchmarks/platform_quality/
python benchmarks/run_all.py --suite platform_quality --profile smoke
